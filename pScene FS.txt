#version 300 es

#ifdef GL_ES
precision highp float;
#endif // GL_ES

layout (location = 0) out vec4 rtFragColor;

in vec2 vTexcoord;

uniform sampler2D uTex;
uniform sampler2D uTex2;

void main()
{
	vec4 col;
	col = mix(texture(uTex, vTexcoord), texture(uTex2, vTexcoord), 0.5);
	rtFragColor = col;
}
